% ── COLORS ────────────────────────────────────────────────────────────────────
\definecolor{thmcolor}{RGB}{0, 70, 127}
\definecolor{defcolor}{RGB}{140, 0, 0}
\definecolor{excolor}{RGB}{30, 100, 30}
\definecolor{refgray}{RGB}{100, 100, 100}

% ── THEOREM ENVIRONMENT STYLES ───────────────────────────────────────────────
\declaretheoremstyle[
  headfont=\bfseries\color{thmcolor},
  notefont=\bfseries,
  notebraces={(}{)},
  bodyfont=\itshape,
  headpunct={.},
  postheadspace=0.5em
]{thmstyle}

\declaretheoremstyle[
  headfont=\bfseries\color{defcolor},
  notefont=\bfseries,
  notebraces={(}{)},
  bodyfont=\normalfont,
  headpunct={.},
  postheadspace=0.5em
]{defstyle}

\declaretheoremstyle[
  headfont=\itshape\color{excolor},
  notefont=\itshape,
  notebraces={(}{)},
  bodyfont=\normalfont,
  headpunct={.},
  postheadspace=0.5em
]{examplestyle}

\declaretheoremstyle[
  headfont=\itshape,
  bodyfont=\normalfont,
  headpunct={.},
  postheadspace=0.5em
]{remarkstyle}

% ── THEOREM ENVIRONMENT DECLARATIONS ─────────────────────────────────────────
\declaretheorem[style=thmstyle,    numberwithin=chapter, name=Theorem]{theorem}
\declaretheorem[style=thmstyle,    sibling=theorem,      name=Lemma]{lemma}
\declaretheorem[style=thmstyle,    sibling=theorem,      name=Corollary]{corollary}
\declaretheorem[style=thmstyle,    sibling=theorem,      name=Proposition]{proposition}
\declaretheorem[style=defstyle,    numberwithin=chapter, name=Definition]{definition}
\declaretheorem[style=examplestyle,numberwithin=chapter, name=Example]{example}
\declaretheorem[style=remarkstyle, numbered=no,          name=Remark]{remark}
\declaretheorem[style=remarkstyle, numbered=no,          name=Note]{note}
