    % Robust real cube root for negative x: sign(x)*|x|^(1/3)
    \addplot[thick, color=defcolor, domain=-2.2:2.2, samples=250] {sign(x)*abs(x)^(1/3)};

    \node[color=thmcolor, anchor=west] at (axis cs:1.4,1.6) {$y=f(x)$};
